2use super::model::SDEModel;
3use rand::Rng;
4use crate::rng;
5use std::f64;
6use rand_distr::{Poisson, Distribution};
7
8pub struct MertonParams {
9    pub s0: f64,
10    pub mu: f64,
11    pub sigma: f64,
12    pub lambda: f64, // Jump intensity
13    pub mu_j: f64,   // Mean of log-jump size
14    pub sigma_j: f64, // Std dev of log-jump size
15}
16
17pub struct Merton {
18    pub params: MertonParams,
19}
20
21impl Merton {
22    pub fn new(params: MertonParams) -> Self {
23        Merton { params }
24    }
25
26    pub fn step<R: Rng + ?Sized>(&self, s: &mut f64, dt: f64, rng: &mut R) {
27        // Continuous part (GBM-like)
28        let z_gbm = rng::get_normal_draw(rng);
29        *s *= ((self.params.mu - 0.5 * self.params.sigma * self.params.sigma) * dt + self.params.sigma * dt.sqrt() * z_gbm).exp();
30
31        // Jump part (Poisson process)
32        let num_jumps: usize = Poisson::new(self.params.lambda * dt).unwrap().sample(rng) as usize;
33        for _ in 0..num_jumps {
34            let jump_size_log = self.params.mu_j + self.params.sigma_j * rng::get_normal_draw(rng);
35            *s *= jump_size_log.exp();
36        }
37    }
38}
39
40impl SDEModel for Merton {
41    // Simplified for generic SDE solvers, focusing on the continuous part.
42    fn drift(&self, s: f64, _t: f64) -> f64 {
43        self.params.mu * s
44    }
45
46    fn diffusion(&self, s: f64, _t: f64) -> f64 {
47        self.params.sigma * s
48    }
49
50    fn diffusion_derivative(&self, _s: f64, _t: f64) -> f64 {
51        self.params.sigma
52    }
53
54    fn step_with_dw(&self, s_current: &mut f64, t_current: f64, dt: f64, dw: f64) {
55        // Simplified 1D step, consistent with the 1D drift and diffusion implementations (continuous part)
56        *s_current += self.drift(*s_current, t_current) * dt + self.diffusion(*s_current, t_current) * dw;
57    }
58}
