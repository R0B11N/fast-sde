1//! Option Payoff Functions
2//! 
3//! # Mathematical Definitions
4//! 
5//! This module implements various option payoff functions that operate on
6//! simulated asset price paths. Each payoff represents a different contract type.
7//! 
8//! ## European Options
9//! - **Call**: max(S_T - K, 0) - right to buy at strike K
10//! - **Put**: max(K - S_T, 0) - right to sell at strike K
11//! 
12//! ## Path-Dependent Options
13//! - **Asian**: Based on average price over the path
14//! - **Barrier**: Knocked out if price crosses barrier level
15//! 
16//! # Implementation Notes
17//! 
18//! All payoffs operate on the full price path `&[f64]` to support
19//! both European (terminal price only) and exotic (full path) options.
20
21use std::f64;
22
23/// Enumeration of supported option payoff types
24/// 
25/// Each variant contains the parameters needed to compute the payoff
26/// from a simulated asset price path.
27#[derive(Clone)]
28pub enum Payoff {
29    /// European call option: max(S_T - K, 0)
30    EuropeanCall { k: f64 },
31    
32    /// European put option: max(K - S_T, 0)  
33    EuropeanPut { k: f64 },
34    
35    /// Asian call option: max(Avg(S_t) - K, 0)
36    AsianCall { k: f64 },
37    
38    /// Up-and-out barrier call: max(S_T - K, 0) if max(S_t) < H, else 0
39    BarrierCallUpAndOut { k: f64, h: f64 },
40    
41    /// Up-and-out barrier put: max(K - S_T, 0) if max(S_t) < H, else 0
42    BarrierPutUpAndOut { k: f64, h: f64 },
43}
44
45impl Payoff {
46    /// Calculate payoff value from a simulated asset price path
47    /// 
48    /// # Parameters
49    /// - `path`: Complete asset price path [S_0, S_1, ..., S_T]
50    /// 
51    /// # Returns
52    /// Non-negative payoff value (options cannot have negative intrinsic value)
53    /// 
54    /// # Mathematical Implementations
55    /// 
56    /// Each payoff type implements its specific mathematical definition:
57    pub fn calculate(&self, path: &[f64]) -> f64 {
58        match self {
59            // European Call: max(S_T - K, 0)
60            // Uses only terminal price (last element of path)
61            Payoff::EuropeanCall { k } => (path.last().unwrap() - k).max(0.0),
62            
63            // European Put: max(K - S_T, 0)
64            // Uses only terminal price (last element of path)
65            Payoff::EuropeanPut { k } => (k - path.last().unwrap()).max(0.0),
66            
67            // Asian Call: max(A - K, 0) where A = (1/n)âˆ‘S_i
68            // Arithmetic average of all prices in the path
69            Payoff::AsianCall { k } => {
70                let average_price: f64 = path.iter().sum::<f64>() / path.len() as f64;
71                (average_price - k).max(0.0)
72            }
73            
74            // Barrier Call Up-and-Out: max(S_T - K, 0) if max(S_t) < H, else 0
75            // Knocked out if price ever touches or exceeds barrier H
76            Payoff::BarrierCallUpAndOut { k, h } => {
77                let mut knocked_out = false;
78                for &price in path {
79                    if price >= *h {
80                        knocked_out = true;
81                        break; // Early termination for efficiency
82                    }
83                }
84                if knocked_out { 0.0 } else { (path.last().unwrap() - k).max(0.0) }
85            }
86            
87            // Barrier Put Up-and-Out: max(K - S_T, 0) if max(S_t) < H, else 0
88            // Knocked out if price ever touches or exceeds barrier H
89            Payoff::BarrierPutUpAndOut { k, h } => {
90                let mut knocked_out = false;
91                for &price in path {
92                    if price >= *h {
93                        knocked_out = true;
94                        break; // Early termination for efficiency
95                    }
96                }
97                if knocked_out { 0.0 } else { (k - path.last().unwrap()).max(0.0) }
98            }
99        }
100    }
101}
