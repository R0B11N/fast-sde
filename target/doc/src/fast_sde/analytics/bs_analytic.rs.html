2//! Analytical Black-Scholes formulas for European options and Greeks
3//! 
4//! # Mathematical Foundation
5//! 
6//! Under the Black-Scholes model, the underlying asset follows:
7//! ```text
8//! dS_t = r S_t dt + σ S_t dW_t
9//! ```
10//! 
11//! The risk-neutral pricing formula gives:
12//! ```text
13//! V(S,t) = e^(-r(T-t)) * E^Q[payoff(S_T) | S_t = S]
14//! ```
15//! 
16//! For European options, this has closed-form solutions involving
17//! the cumulative normal distribution function Φ(x).
18
19use std::f64::consts::PI;
20use crate::math_utils::norm_cdf;
21
22/// Standard normal probability density function
23/// 
24/// # Formula
25/// ```text
26/// φ(x) = (1/√(2π)) * exp(-x²/2)
27/// ```
28fn norm_pdf(x: f64) -> f64 {
29    (1.0 / (2.0 * PI).sqrt()) * (-0.5 * x * x).exp()
30}
31
32/// Black-Scholes European call option price
33/// 
34/// # Formula
35/// ```text
36/// C(S,K,r,σ,T) = S*Φ(d₁) - K*e^(-rT)*Φ(d₂)
37/// ```
38/// 
39/// Where:
40/// ```text
41/// d₁ = [ln(S/K) + (r + σ²/2)T] / (σ√T)
42/// d₂ = d₁ - σ√T
43/// ```
44/// 
45/// # Parameters
46/// - `s`: Current stock price
47/// - `k`: Strike price  
48/// - `r`: Risk-free rate
49/// - `sigma`: Volatility
50/// - `t`: Time to expiration
51/// 
52/// # Returns
53/// Present value of the call option
54pub fn bs_call_price(s: f64, k: f64, r: f64, sigma: f64, t: f64) -> f64 {
55    let d1 = ( (s / k).ln() + (r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
56    let d2 = d1 - sigma * t.sqrt();
57    s * norm_cdf(d1) - k * (-r * t).exp() * norm_cdf(d2)
58}
59
60/// Black-Scholes European put option price
61/// 
62/// # Formula
63/// ```text
64/// P(S,K,r,σ,T) = K*e^(-rT)*Φ(-d₂) - S*Φ(-d₁)
65/// ```
66/// 
67/// Uses put-call parity relationship with call price.
68pub fn bs_put_price(s: f64, k: f64, r: f64, sigma: f64, t: f64) -> f64 {
69    let d1 = ( (s / k).ln() + (r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
70    let d2 = d1 - sigma * t.sqrt();
71    k * (-r * t).exp() * norm_cdf(-d2) - s * norm_cdf(-d1)
72}
73
74/// Black-Scholes Delta (∂V/∂S) for European call
75/// 
76/// # Formula
77/// ```text
78/// Δ = ∂C/∂S = Φ(d₁)
79/// ```
80/// 
81/// # Interpretation
82/// - Hedge ratio: number of shares to buy per option sold
83/// - Probability of finishing in-the-money (risk-neutral measure)
84/// - Range: [0, 1] for calls
85pub fn bs_call_delta(s: f64, k: f64, r: f64, sigma: f64, t: f64) -> f64 {
86    let d1 = ( (s / k).ln() + (r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
87    norm_cdf(d1)
88}
89
90/// Black-Scholes Gamma (∂²V/∂S²) for European call
91/// 
92/// # Formula
93/// ```text
94/// Γ = ∂²C/∂S² = φ(d₁) / (S * σ * √T)
95/// ```
96/// 
97/// # Interpretation
98/// - Rate of change of Delta w.r.t. underlying price
99/// - Convexity of option price
100/// - Maximum at-the-money, decreases as option goes in/out-of-money
101/// - Same for calls and puts
102pub fn bs_call_gamma(s: f64, _k: f64, _r: f64, sigma: f64, t: f64) -> f64 {
103    let d1 = ( (s / _k).ln() + (_r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
104    norm_pdf(d1) / (s * sigma * t.sqrt())
105}
106
107/// Black-Scholes Vega (∂V/∂σ) for European call
108/// 
109/// # Formula
110/// ```text
111/// ν = ∂C/∂σ = S * φ(d₁) * √T
112/// ```
113/// 
114/// # Interpretation
115/// - Sensitivity to volatility changes
116/// - Always positive for long options
117/// - Maximum at-the-money
118/// - Units: price change per 1% volatility change
119pub fn bs_call_vega(s: f64, _k: f64, _r: f64, sigma: f64, t: f64) -> f64 {
120    let d1 = ( (s / _k).ln() + (_r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
121    s * norm_pdf(d1) * t.sqrt()
122}
123
124/// Black-Scholes Theta (∂V/∂t) for European call
125/// 
126/// # Formula
127/// ```text
128/// Θ = ∂C/∂t = -S*φ(d₁)*σ/(2√T) - r*K*e^(-rT)*Φ(d₂)
129/// ```
130/// 
131/// # Interpretation
132/// - Time decay of option value
133/// - Usually negative for long options (time erodes value)
134/// - Accelerates as expiration approaches
135/// - Units: price change per day
136pub fn bs_call_theta(s: f64, k: f64, r: f64, sigma: f64, t: f64) -> f64 {
137    let d1 = ( (s / k).ln() + (r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
138    let d2 = d1 - sigma * t.sqrt();
139    (-s * norm_pdf(d1) * sigma) / (2.0 * t.sqrt()) - r * k * (-r * t).exp() * norm_cdf(d2)
140}
141
142/// Black-Scholes Rho (∂V/∂r) for European call
143/// 
144/// # Formula
145/// ```text
146/// ρ = ∂C/∂r = K * T * e^(-rT) * Φ(d₂)
147/// ```
148/// 
149/// # Interpretation
150/// - Sensitivity to interest rate changes
151/// - Positive for calls (higher rates increase call value)
152/// - Negative for puts
153/// - Units: price change per 1% rate change
154pub fn bs_call_rho(s: f64, k: f64, r: f64, sigma: f64, t: f64) -> f64 {
155    let d1 = ( (s / k).ln() + (r + 0.5 * sigma * sigma) * t ) / (sigma * t.sqrt());
156    let d2 = d1 - sigma * t.sqrt();
157    k * t * (-r * t).exp() * norm_cdf(d2)
158}
