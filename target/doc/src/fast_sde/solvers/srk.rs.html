2//! Stochastic Runge-Kutta (SRK) Scheme for SDE Integration
3//! 
4//! # Mathematical Framework
5//! 
6//! Stochastic analog of the classical Runge-Kutta method for ODEs.
7//! This implementation uses a Heun-like predictor-corrector approach.
8//! 
9//! # Algorithm
10//! 
11//! For SDE: `dX_t = a(X_t, t) dt + b(X_t, t) dW_t`
12//! 
13//! 1. **Predictor step** (Euler):
14//!    ```text
15//!    X* = X_n + a(X_n, t_n) Δt + b(X_n, t_n) ΔW_n
16//!    ```
17//! 
18//! 2. **Corrector step** (average):
19//!    ```text
20//!    X_{n+1} = X_n + ½[a(X_n, t_n) + a(X*, t_{n+1})] Δt + ½[b(X_n, t_n) + b(X*, t_{n+1})] ΔW_n
21//!    ```
22//! 
23//! # Convergence Properties
24//! 
25//! - **Strong convergence**: Order 1.0 
26//! - **Weak convergence**: Order 1.0
27//! - **Cost**: 2x function evaluations per step (vs Euler)
28//! 
29//! # Advantages
30//! 
31//! - Higher accuracy than Euler-Maruyama
32//! - More stable for stiff SDEs
33//! - Good for models with state-dependent coefficients
34
35use crate::models::model::SDEModel;
36use rand::Rng;
37use crate::rng;
38use std::f64;
39
40/// Stochastic Runge-Kutta numerical scheme
41pub struct Srk;
42
43impl Srk {
44    pub fn new() -> Self {
45        Srk {}
46    }
47
48    /// Single SRK step using predictor-corrector method
49    /// 
50    /// # Algorithm Details
51    /// 
52    /// 1. **Predictor**: Use Euler step to estimate X* at t+Δt
53    /// 2. **Corrector**: Average drift and diffusion at both time points
54    /// 3. **Same random increment**: Use same ΔW for both steps
55    /// 
56    /// # Performance vs Accuracy
57    /// 
58    /// - 2x slower than Euler (requires 2 function evaluations)
59    /// - Significantly more accurate, especially for larger time steps
60    /// - Good choice when function evaluations are cheap
61    pub fn step<M: SDEModel, R: Rng + ?Sized>(model: &M, s: &mut f64, t: f64, dt: f64, rng: &mut R) {
62        let dw = dt.sqrt() * rng::get_normal_draw(rng);
63
64        // Predictor step: Euler-Maruyama to get provisional value
65        let s_star = *s + model.drift(*s, t) * dt + model.diffusion(*s, t) * dw;
66
67        // Corrector step: average coefficients between start and end points
68        let drift_avg = 0.5 * (model.drift(*s, t) + model.drift(s_star, t + dt));
69        let diffusion_avg = 0.5 * (model.diffusion(*s, t) + model.diffusion(s_star, t + dt));
70
71        // Apply corrected update using averaged coefficients
72        *s += drift_avg * dt + diffusion_avg * dw;
73    }
74}
